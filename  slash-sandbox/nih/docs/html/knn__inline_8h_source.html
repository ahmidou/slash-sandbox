00035 template <typename VectorType, typename PointIterator>
00036 __device__ void lookup(
00037     const VectorType        query,
00038     const Kd_node*          kd_nodes,
00039     const uint2*            kd_ranges,
00040     const uint2*            kd_leaves,
00041     const PointIterator     kd_points,
00042     Kd_knn<3>::Result*      results)
00043 {
00044     //
00045     // 1-st pass: find the leaf containing the query point and compute an upper bound
00046     // on the search distance
00047     //
00048 
00049     uint32 idx;
00050     float  dist2 = 1.0e16f;
00051 
00052     // start from the root node
00053     uint32 node_index = 0;
00054 
00055     // keep track of which leaf we visited
00056     uint32 first_leaf = 0;
00057 
00058     while (1)
00059     {
00060         const Kd_node node = kd_nodes[ node_index ];
00061 
00062         if (node.is_leaf())
00063         {
00064             // find the closest neighbor in this leaf
00065             const uint2 leaf = kd_leaves[ node.get_leaf_index() ];
00066             for (uint32 i = leaf.x; i < leaf.y; ++i)
00067             {
00068                 const VectorType delta = kd_points[i] - query;
00069                 const float d2 = delta[0]*delta[0] + delta[1]*delta[1] + delta[2]*delta[2];
00070                 if (dist2 > d2)
00071                 {
00072                     dist2 = d2;
00073                     idx   = i;
00074                 }
00075             }
00076 
00077             // keep track of which leaf we found
00078             first_leaf = node_index;
00079             break;
00080         }
00081         else
00082         {
00083             const uint32 split_dim   = node.get_split_dim();
00084             const float  split_plane = node.get_split_plane();
00085 
00086             node_index =
00087                 split_dim == 0 ? node.get_child_offset() + (query[0] < split_plane ? 0u : 1u) :
00088                 split_dim == 1 ? node.get_child_offset() + (query[1] < split_plane ? 0u : 1u) :
00089                                  node.get_child_offset() + (query[2] < split_plane ? 0u : 1u);
00090         }
00091     }
00092 
00093     //
00094     // 2-nd pass: visit the tree with a stack and careful pruning
00095     //
00096 
00097     int32  stackp = 1;
00098     float4 stack[32];
00099 
00100     // place a sentinel node in the stack
00101     stack[0] = make_float4( 0.0f, 0.0f, 0.0f, binary_cast<float>(uint32(-1)) );
00102 
00103     // start from the root node
00104     node_index = 0;
00105 
00106     float3 cdist = make_float3( 0.0f, 0.0f, 0.0f );
00107 
00108     while (node_index != uint32(-1))
00109     {
00110         const Kd_node node = kd_nodes[ node_index ];
00111 
00112         if (node.is_leaf())
00113         {
00114             if (first_leaf != node_index)
00115             {
00116                 // find the closest neighbor in this leaf
00117                 const uint2 leaf = kd_leaves[ node.get_leaf_index() ];
00118                 for (uint32 i = leaf.x; i < leaf.y; ++i)
00119                 {
00120                     const VectorType delta = kd_points[i] - query;
00121                     const float d2 = delta[0]*delta[0] + delta[1]*delta[1] + delta[2]*delta[2];
00122                     if (dist2 > d2)
00123                     {
00124                         dist2 = d2;
00125                         idx   = i;
00126                     }
00127                 }
00128             }
00129 
00130             // pop the next node from the stack
00131             while (stackp > 0)
00132             {
00133                 const float4 stack_node = stack[ --stackp ];
00134                 node_index = binary_cast<uint32>( stack_node.w );
00135                 cdist      = make_float3( stack_node.x, stack_node.y, stack_node.z );
00136 
00137                 if (cdist.x*cdist.x + cdist.y*cdist.y + cdist.z*cdist.z > dist2)
00138                     continue;
00139             }
00140         }
00141         else
00142         {
00143             const uint32 split_dim   = node.get_split_dim();
00144             const float  split_plane = node.get_split_plane();
00145 
00146             const float split_dist = 
00147                 split_dim == 0 ? (query[0] - split_plane) :
00148                 split_dim == 1 ? (query[1] - split_plane) :
00149                                  (query[2] - split_plane);
00150 
00151             const uint32 select = split_dist <= 0.0f ? 0u : 1u;
00152 
00153             node_index = node.get_child_offset() + select;
00154 
00155             // compute the vector distance to the far node
00156             float3 cdist_far = cdist;
00157             if (split_dim == 0)      cdist_far.x = split_dist;
00158             else if (split_dim == 1) cdist_far.y = split_dist;
00159             else                     cdist_far.z = split_dist;
00160 
00161             // check whether we should push the far node on the stack
00162             const float dist_far2 = cdist_far.x*cdist_far.x +
00163                                     cdist_far.y*cdist_far.y +
00164                                     cdist_far.z*cdist_far.z;
00165 
00166             if (dist_far2 <= dist2)
00167             {
00168                 stack[ stackp++ ] = make_float4(
00169                     cdist_far.x,
00170                     cdist_far.y,
00171                     cdist_far.z,
00172                     binary_cast<float>( node.get_child_offset() + 1u - select ) );
00173             }
00174         }
00175     }
00176 
00177     // write the result
00178     results->index = idx;
00179     results->dist2 = dist2;
00180 }

00182 template <typename QueryIterator, typename PointIterator>
00183 __global__ void lookup_kernel(
00184     const uint32                    n_points,
00185     const QueryIterator             points_begin,
00186     const Kd_node*                  kd_nodes,
00187     const uint2*                    kd_ranges,
00188     const uint2*                    kd_leaves,
00189     const PointIterator             kd_points,
00190     Kd_knn<3>::Result*              results)
00191 {
00192     const uint32 grid_size = gridDim.x * blockDim.x;
00193 
00194     // loop through all logical blocks associated to this physical one
00195     for (uint32 base_idx = blockIdx.x * blockDim.x;
00196                 base_idx < n_points;
00197                 base_idx += grid_size)
00198     {
00199         const uint32 index = threadIdx.x + base_idx;
00200         if (index >= n_points)
00201             return;
00202 
00203         lookup(
00204             thrust::detail::backend::dereference( points_begin + index ),
00205             kd_nodes,
00206             kd_ranges,
00207             kd_leaves,
00208             kd_points,
00209             results + index );
00210     }
00211 }

00215 // perform a k-nn lookup for a set of query points
00216 //
00217 // \param points_begin     beginning of the query point sequence
00218 // \param points_end       end of the query point sequence
00219 // \param kd_nodes         k-d tree nodes
00220 // \param kd_ranges        k-d tree node ranges
00221 // \param kd_leaves        k-d tree leaves
00222 // \param kd_points        k-d tree points
00223 template <typename QueryIterator, typename PointIterator>
00224 void Kd_knn<3>::run(
00225     const QueryIterator             points_begin,
00226     const QueryIterator             points_end,
00227     const Kd_node*                  kd_nodes,
00228     const uint2*                    kd_ranges,
00229     const uint2*                    kd_leaves,
00230     const PointIterator             kd_points,
00231     Result*                         results)
00232 {
00233     const uint32 n_points = uint32( points_end - points_begin );
00234 
00235     const uint32 BLOCK_SIZE = 128;
00236     const size_t max_blocks = thrust::detail::backend::cuda::arch::max_active_blocks(
00237         kd_knn::lookup_kernel<QueryIterator,PointIterator>, BLOCK_SIZE, 0);
00238     const size_t n_blocks   = nih::min( max_blocks, (n_points + BLOCK_SIZE-1) / BLOCK_SIZE );
00239 
00240     kd_knn::lookup_kernel<<<n_blocks,BLOCK_SIZE>>> (
00241         n_points,
00242         points_begin,
00243         kd_nodes,
00244         kd_ranges,
00245         kd_leaves,
00246         kd_points,
00247         results );
00248 
00249     cudaThreadSynchronize();
00250 }
