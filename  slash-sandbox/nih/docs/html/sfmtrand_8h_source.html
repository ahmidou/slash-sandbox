00107                 static __m128i mm_recursion(__m128i *a, const __m128i* b, __m128i c,
00108                         __m128i d, __m128i mask) {
00109                         __m128i v, x, y, z;
00110                         x = _mm_load_si128(a);
00111                         y = _mm_srli_epi32(*b, SR1);
00112                         z = _mm_srli_si128(c, SR2);
00113                         v = _mm_slli_epi32(d, SL1);
00114                         z = _mm_xor_si128(z, x);
00115                         z = _mm_xor_si128(z, v);
00116                         x = _mm_slli_si128(x, SL2);
00117                         y = _mm_and_si128(y, mask);
00118                         z = _mm_xor_si128(z, x);
00119                         z = _mm_xor_si128(z, y);
00120                         return z;

00122                 void period_certification(void) {
00123                         int inner = 0;
00124                         int i, j;
00125                         uint32_t work;
00126 
00127                         for (i = 0; i < 4; i++)
00128                                 inner ^= psfmt32[idxof(i)] & parity[i];
00129                         for (i = 16; i > 0; i >>= 1)
00130                                 inner ^= inner >> i;
00131                         inner &= 1;
00132                         /* check OK */
00133                         if (inner == 1) {
00134                                 return;
00135                         }
00136                         /* check NG, and modification */
00137                         for (i = 0; i < 4; i++) {
00138                                 work = 1;
00139                                 for (j = 0; j < 32; j++) {
00140                                         if ((work & parity[i]) != 0) {
00141                                                 psfmt32[idxof(i)] ^= work;
00142                                                 return;
00143                                         }
00144                                         work = work << 1;
00145                                 }
00146                         }

00239                 void init_by_array(uint32_t *init_key, int key_length) {
00240                         int i, j, count;
00241                         uint32_t r;
00242                         int lag;
00243                         int mid;
00244                         int size = N * 4;
00245 
00246                         if (size >= 623) {
00247                                 lag = 11;
00248                         } else if (size >= 68) {
00249                                 lag = 7;
00250                         } else if (size >= 39) {
00251                                 lag = 5;
00252                         } else {
00253                                 lag = 3;
00254                         }
00255                         mid = (size - lag) / 2;
00256 
00257                         memset(sfmt, 0x8b, sizeof(sfmt));
00258                         if (key_length + 1 > N32) {
00259                                 count = key_length + 1;
00260                         } else {
00261                                 count = N32;
00262                         }
00263                         r = func1(psfmt32[idxof(0)] ^ psfmt32[idxof(mid)] ^ psfmt32[idxof(N32 - 1)]);
00264                         psfmt32[idxof(mid)] += r;
00265                         r += key_length;
00266                         psfmt32[idxof(mid + lag)] += r;
00267                         psfmt32[idxof(0)] = r;
00268 
00269                         count--;
00270                         for (i = 1, j = 0; (j < count) && (j < key_length); j++) {
00271                                 r = func1(psfmt32[idxof(i)] ^ psfmt32[idxof((i + mid) % N32)]
00272                                         ^ psfmt32[idxof((i + N32 - 1) % N32)]);
00273                                 psfmt32[idxof((i + mid) % N32)] += r;
00274                                 r += init_key[j] + i;
00275                                 psfmt32[idxof((i + mid + lag) % N32)] += r;
00276                                 psfmt32[idxof(i)] = r;
00277                                 i = (i + 1) % N32;
00278                         }
00279                         for (; j < count; j++) {
00280                                 r = func1(psfmt32[idxof(i)] ^ psfmt32[idxof((i + mid) % N32)]
00281                                         ^ psfmt32[idxof((i + N32 - 1) % N32)]);
00282                                 psfmt32[idxof((i + mid) % N32)] += r;
00283                                 r += i;
00284                                 psfmt32[idxof((i + mid + lag) % N32)] += r;
00285                                 psfmt32[idxof(i)] = r;
00286                                 i = (i + 1) % N32;
00287                         }
00288                         for (j = 0; j < N32; j++) {
00289                                 r = func2(psfmt32[idxof(i)] + psfmt32[idxof((i + mid) % N32)]
00290                                         + psfmt32[idxof((i + N32 - 1) % N32)]);

00292                                 r -= i;
