00062 // Helper class to collect the 8 children of a given node from
00063 // a binary kd-tree structure.
00064 // The class implements template based compile-time recursion.
00065 template <uint32 LEVEL, uint32 STRIDE>
00066 struct Octree_collector
00067 {
00068     static NIH_HOST_DEVICE void find_children(
00069         const uint32        node_index,
00070         const Kd_node*      nodes,
00071         Octree_collection*  result,
00072         uint32*             children,
00073         const uint32        octant = 0)
00074     {
00075         const Kd_node node = nodes[ node_index ];
00076 
00077         const bool active0 = node.has_child(0);
00078         const bool active1 = node.has_child(1);
00079 
00080         if ((active0 == false) && (active1 == false))
00081         {
00082             // here we have some trouble... this guy is likely not
00083             // fitting within an octant of his parent.
00084             // Which octant do we assign this guy to?
00085             // Let's say the minimum...
00086             children[ STRIDE * result->node_count ] = node_index;
00087             result->bitmask |= 1u << (octant << (3 - LEVEL));
00088             result->node_count += 1;
00089             result->leaf_count += 1;
00090         }
00091         else
00092         {
00093             // traverse the children in Morton order: preserving
00094             // the order is key here, as we want the output counter
00095             // to match the bitmask pop-counts.
00096             if (active0)
00097             {
00098                 Octree_collector<LEVEL+1,STRIDE>::find_children(
00099                     node.get_left(),
00100                     nodes,
00101                     result,
00102                     children,
00103                     octant * 2 );
00104             }
00105             if (active1)
00106             {
00107                 Octree_collector<LEVEL+1,STRIDE>::find_children(
00108                     node.get_right(),
00109                     nodes,
00110                     result,
00111                     children,
00112                     octant * 2 + 1 );
00113             }
00114         }
00115     }
00116 };
00117 // Terminal node of Octree_collector's compile-time recursion.
00118 template <uint32 STRIDE>
00119 struct Octree_collector<3,STRIDE>
00120 {
00121     static NIH_HOST_DEVICE void find_children(
00122         const uint32        node_index,
00123         const Kd_node*      nodes,
00124         Octree_collection*  result,
00125         uint32*             children,
00126         const uint32        octant)
00127     {
00128         // we got to one of the octants
00129         children[ STRIDE * result->node_count ] = node_index;
00130         result->bitmask |= 1u << octant;
00131         result->node_count += 1;
00132 
00133         if (nodes[ node_index ].is_leaf())
00134             result->leaf_count += 1;
00135     }

00138 // collect octants from a kd-tree
00139 template <uint32 BLOCK_SIZE>
00140 __global__ void collect_octants_kernel(
00141     const uint32        grid_size,
00142     const Kd_node*      kd_nodes,
00143     const uint32        in_tasks_count,
00144     const Split_task*   in_tasks,
00145     uint32*             out_tasks_count,
00146     Split_task*         out_tasks,
00147     uint32*             out_nodes_count,
00148     Octree_node*   out_nodes)
00149 {
00150     const uint32 LOG_WARP_SIZE = 5;
00151     const uint32 WARP_SIZE = 1u << LOG_WARP_SIZE;
00152 
00153     volatile __shared__ uint32 warp_offset[ BLOCK_SIZE >> LOG_WARP_SIZE ];
00154 
00155     const uint32 warp_tid = threadIdx.x & (WARP_SIZE-1);
00156     const uint32 warp_id  = threadIdx.x >> LOG_WARP_SIZE;
00157 
00158     volatile __shared__ uint32 sm_red[ BLOCK_SIZE * 2 ];
00159     volatile uint32* warp_red = sm_red + WARP_SIZE * 2 * warp_id;
00160 
00161     __shared__ uint32 sm_children[ BLOCK_SIZE * 8 ];
00162     uint32* children = sm_children + threadIdx.x;
00163 
00164     // loop through all logical blocks associated to this physical one
00165     for (uint32 base_idx = blockIdx.x * BLOCK_SIZE;
00166         base_idx < in_tasks_count;
00167         base_idx += grid_size)
00168     {
00169         const uint32 task_id = threadIdx.x + base_idx;
00170 
00171         uint32 node;
00172 
00173         Octree_collection result;
00174         result.node_count = 0;
00175         result.leaf_count = 0;
00176         result.bitmask    = 0;
00177 
00178         // check if the task id is in range, and if so try to collect its treelet
00179         if (task_id < in_tasks_count)
00180         {
00181             const Split_task in_task = in_tasks[ task_id ];
00182 
00183             node = in_task.m_node;
00184 
00185             Octree_collector<0,BLOCK_SIZE>::find_children(
00186                 in_task.m_input,
00187                 kd_nodes,
00188                 &result,
00189                 children );
00190         }
00191 
00192         // allocate output nodes, output tasks, and write all leaves
00193         {
00194             uint32 task_count = result.node_count - result.leaf_count;
00195 
00196             uint32 node_offset = cuda::alloc( result.node_count, out_nodes_count, warp_tid, warp_red, warp_offset + warp_id );
00197             uint32 task_offset = cuda::alloc( task_count,        out_tasks_count, warp_tid, warp_red, warp_offset + warp_id );
00198 
00199             // write the parent node
00200             if (task_id < in_tasks_count)
00201                 out_nodes[ node ] = Octree_node( result.bitmask, node_offset );
00202 
00203             // write out all outputs
00204             for (uint32 i = 0; i < result.node_count; ++i)
00205             {
00206                 const uint32  kd_node_index = children[ i * BLOCK_SIZE ];
00207                 const Kd_node kd_node       = kd_nodes[ kd_node_index ];
00208 
00209                 if (kd_node.is_leaf() == false)
00210                     out_tasks[ task_offset++ ] = Split_task( node_offset, 0, 0, kd_node_index );
00211                 else
00212                     out_nodes[ node_offset ] = Octree_node( kd_node.get_child_offset() );
00213 
00214                 node_offset++;
00215             }
00216         }
00217     }
00218 }

00221 inline void collect_octants(
00222     const Kd_node*      kd_nodes,
00223     const uint32        in_tasks_count,
00224     const Split_task*   in_tasks,

00226     Split_task*         out_tasks,
00227     uint32*             out_nodes_count,
00228     Octree_node*   out_nodes)

00231     const size_t max_blocks = thrust::detail::device::cuda::arch::max_active_blocks(collect_octants_kernel<BLOCK_SIZE>, BLOCK_SIZE, 0);
00232     const size_t n_blocks   = nih::min( max_blocks, (in_tasks_count + BLOCK_SIZE-1) / BLOCK_SIZE );

00236         grid_size,

00241         out_tasks,

00254     const Bbox3f                           bbox,
00255     const Iterator                         points_begin,
00256     const Iterator                         points_end,

00271         points_begin,
00272         points_begin + n_points,
00273         m_codes.begin(),
00274         morton_functor<Integer>( bbox ) );

00276     // setup the point indices, from 0 to n_points-1
00277     thrust::copy(
00278         thrust::counting_iterator<uint32>(0),
00279         thrust::counting_iterator<uint32>(0) + n_points,
00280         m_index->begin() );

00282     // sort the indices by Morton code
00283     // TODO: use Duane's library directly here... this is doing shameful allocations!
00284     thrust::sort_by_key(
00285         m_codes.begin(),
00286         m_codes.begin() + n_points,
00287         m_index->begin() );
00288 
00289     // generate a kd-tree
00290     cuda::Bintree_context tree( m_kd_nodes, *m_leaves );
00291 
00292     cuda::generate(
00293         m_kd_context,
00294         n_points,
00295         thrust::raw_pointer_cast( &m_codes.front() ),
00296         octree_builder::Morton_bits<Integer>::value,
00297         max_leaf_size,
00298         true,
00299         tree );
00300 
00301     m_leaf_count = m_kd_context.m_leaves;
00302 
00303     // start building the octree
00304     thrust::device_vector<Split_task>* m_task_queues = m_kd_context.m_task_queues;
00305     thrust::device_vector<uint32>&     m_counters    = m_kd_context.m_counters;
00306 
00307     need_space( m_task_queues[0], n_points );
00308     need_space( m_task_queues[1], n_points );
00309 
00310     Split_task* task_queues[2] = {
00311         thrust::raw_pointer_cast( &(m_task_queues[0]).front() ),
00312         thrust::raw_pointer_cast( &(m_task_queues[1]).front() )
00313     };
00314 
00315     uint32 in_queue  = 0;
00316     uint32 out_queue = 1;
00317 
00318     // convert the kd-tree into an octree.
00319     need_space( m_counters, 3 );
00320     m_counters[ in_queue ]  = 1;
00321     m_counters[ out_queue ] = 0;
00322     m_counters[ 2 ]         = 1; // output node counter
00323 
00324     m_task_queues[ in_queue ][0] = Split_task( 0, 0, 0, 0 );
00325 
00326     uint32 level = 0;
00327     m_levels[ level++ ] = 0;
00328 
00329     // loop until there's tasks left in the input queue
00330     while (m_counters[ in_queue ])
00331     {
00332         m_levels[ level++ ] = m_counters[2];
00333 
00334         need_space( *m_octree, m_counters[2] + m_counters[ in_queue ]*8 );
00335 
00336         // clear the output queue
00337         m_counters[ out_queue ] = 0;
00338         cudaThreadSynchronize();
00339 
00340         octree_builder::collect_octants(
00341             thrust::raw_pointer_cast( &m_kd_nodes.front() ),
00342             m_counters[ in_queue ],
00343             task_queues[ in_queue ],
00344             thrust::raw_pointer_cast( &m_counters.front() ) + out_queue,
00345             task_queues[ out_queue ],
00346             thrust::raw_pointer_cast( &m_counters.front() ) + 2,
00347             thrust::raw_pointer_cast( &m_octree->front() ) );
00348 
00349         const uint32 out_queue_size = m_counters[ out_queue ];
00350 
00351         // swap the input and output queues
00352         std::swap( in_queue, out_queue );
00353 
00354         const uint32 n_nodes = m_counters[2];
00355     }
00356     m_node_count = m_counters[2];

00359         m_levels[ level ] = m_node_count;
