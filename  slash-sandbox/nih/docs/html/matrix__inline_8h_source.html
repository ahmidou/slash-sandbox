00228 template <typename T>
00229 bool invert(const Matrix<T,4,4>& a, Matrix<T,4,4>& r)
00230 {
00231         T t1, t2, t3, t4, t5, t6;
00232 
00233         // We calculate the adjoint matrix of a, then we divide it by the determinant of a
00234         // We indicate with Aij the cofactor of a[i][j]:   Aij = (-1)^(i+j)*Det(Tij), where
00235         // Tij is the minor complementary of a[i][j]
00236 
00237         // First block ([0,0] - [3,1])
00238 
00239         t1 = a(2,2)*a(3,3) - a(2,3)*a(3,2);
00240         t2 = a(2,1)*a(3,3) - a(2,3)*a(3,1);
00241         t3 = a(2,1)*a(3,2) - a(2,2)*a(3,1);
00242         t4 = a(2,0)*a(3,3) - a(2,3)*a(3,0);
00243         t5 = a(2,0)*a(3,2) - a(2,2)*a(3,0);
00244         t6 = a(2,0)*a(3,1) - a(2,1)*a(3,0);
00245 
00246         r[0][0] =   a(1,1) * t1 - a(1,2) * t2 + a(1,3) * t3;      // A00
00247         r[0][1] = -(a(0,1) * t1 - a(0,2) * t2 + a(0,3) * t3);     // A10
00248         r[1][0] = -(a(1,0) * t1 - a(1,2) * t4 + a(1,3) * t5);     // A01
00249         r[1][1] =   a(0,0) * t1 - a(0,2) * t4 + a(0,3) * t5;      // A11
00250         r[2][0] =   a(1,0) * t2 - a(1,1) * t4 + a(1,3) * t6;      // A02
00251         r[2][1] = -(a(0,0) * t2 - a(0,1) * t4 + a(0,3) * t6);     // A21
00252         r[3][0] = -(a(1,0) * t3 - a(1,1) * t5 + a(1,2) * t6);     // A03
00253         r[3][1] =   a(0,0) * t3 - a(0,1) * t5 + a(0,2) * t6;      // A13
00254 
00255         // Second block ([0,2] - [3,2])
00256 
00257         t1 = a(1,2)*a(3,3) - a(1,3)*a(3,2);
00258         t2 = a(1,1)*a(3,3) - a(1,3)*a(3,1);
00259         t3 = a(1,1)*a(3,2) - a(1,2)*a(3,1);
00260         t4 = a(1,0)*a(3,3) - a(1,3)*a(3,0);
00261         t5 = a(1,0)*a(3,2) - a(1,2)*a(3,0);
00262         t6 = a(1,0)*a(3,1) - a(1,1)*a(3,0);
00263 
00264         r[0][2] =   a(0,1) * t1 - a(0,2) * t2 + a(0,3) * t3;      // A20
00265         r[1][2] = -(a(0,0) * t1 - a(0,2) * t4 + a(0,3) * t5);     // A21
00266         r[2][2] =   a(0,0) * t2 - a(0,1) * t4 + a(0,3) * t6;      // A22
00267         r[3][2] = -(a(0,0) * t3 - a(0,1) * t5 + a(0,2) * t6);     // A23
00268 
00269         // Third block ([0,3] - [3,3])
00270 
00271         t1 = a(1,2)*a(2,3) - a(1,3)*a(2,2);
00272         t2 = a(1,1)*a(2,3) - a(1,3)*a(2,1);
00273         t3 = a(1,1)*a(2,2) - a(1,2)*a(2,1);
00274         t4 = a(1,0)*a(2,3) - a(1,3)*a(2,0);
00275         t5 = a(1,0)*a(2,2) - a(1,2)*a(2,0);
00276         t6 = a(1,0)*a(2,1) - a(1,1)*a(2,0);
00277 
00278         r[0][3] = -(a(0,1) * t1 - a(0,2) * t2 + a(0,3) * t3);     // A30
00279         r[1][3] =   a(0,0) * t1 - a(0,2) * t4 + a(0,3) * t5;      // A31
00280         r[2][3] = -(a(0,0) * t2 - a(0,1) * t4 + a(0,3) * t6);     // A32
00281         r[3][3] =   a(0,0) * t3 - a(0,1) * t5 + a(0,2) * t6;      // A33
00282 
00283         // We save some time calculating Det(a) this way (now r is adjoint of a)
00284         // Det(a) = a00 * A00 + a01 * A01 + a02 * A02 + a03 * A03
00285         T d = a(0,0)*r[0][0] + a(0,1)*r[1][0] + a(0,2)*r[2][0] + a(0,3)*r[3][0];
00286 
00287         if (d == T(0.0)) return false; // Singular matrix => no inverse
00288 
00289         d = T(1.0)/d;
00290 
00291         r *= d;
00292         return true;
00293 }
