00237 template <typename T>
00238 NIH_HOST NIH_DEVICE Vector<T,3> orthogonal(const Vector<T,3> v)
00239 {
00240     if (v[0]*v[0] < v[1]*v[1])
00241         {
00242                 if (v[0]*v[0] < v[2]*v[2])
00243                 {
00244                         // r = -cross( v, (1,0,0) )
00245                         return Vector<T,3>( 0.0f, -v[2], v[1] );
00246                 }
00247                 else
00248                 {
00249                         // r = -cross( v, (0,0,1) )
00250                         return Vector<T,3>( -v[1], v[0], 0.0 );
00251                 }
00252         }
00253         else
00254         {
00255                 if (v[1]*v[1] < v[2]*v[2])
00256                 {
00257                         // r = -cross( v, (0,1,0) )
00258                         return Vector<T,3>( v[2], 0.0, -v[0] );
00259                 }
00260                 else
00261                 {
00262                         // r = -cross( v, (0,0,1) )
00263                         return Vector<T,3>( -v[1], v[0], 0.0 );
