00078 FORCE_INLINE NIH_DEVICE void update_bin(float3* bin_bmin, float3* bin_bmax, int32* bin_counter, const Bbox4f bbox)
00079 {
00080     const float3 bmin = *bin_bmin;
00081     if (bbox[0][0] < bmin.x) atomicMin( (int32*)&(bin_bmin->x), __float_as_int(bbox[0][0]) );
00082     if (bbox[0][1] < bmin.y) atomicMin( (int32*)&(bin_bmin->y), __float_as_int(bbox[0][1]) );
00083     if (bbox[0][2] < bmin.z) atomicMin( (int32*)&(bin_bmin->z), __float_as_int(bbox[0][2]) );
00084 
00085     const float3 bmax = *bin_bmax;
00086     if (bbox[1][0] > bmax.x) atomicMax( (int32*)&(bin_bmax->x), __float_as_int(bbox[1][0]) );
00087     if (bbox[1][1] > bmax.y) atomicMax( (int32*)&(bin_bmax->y), __float_as_int(bbox[1][1]) );
00088     if (bbox[1][2] > bmax.z) atomicMax( (int32*)&(bin_bmax->z), __float_as_int(bbox[1][2]) );
00089 
00090     atomicAdd( bin_counter, 1 );
00091 }

00111                 base_idx < n_objects;

00125       #endif

00134             const Bin node_bbox = queue.bins[split_id];

00136             node_bbox.bmax.x - node_bbox.bmin.x,
00137             node_bbox.bmax.y - node_bbox.bmin.y,
00138             node_bbox.bmax.z - node_bbox.bmin.z );
00139 
00140         Bbox4f bbox = bboxes[id];
00141         bbox[0] += origin;
00142         bbox[1] += origin;
00143         const Vector3f center = (xyz(bbox[0]) + xyz(bbox[1]))*0.5f;
00144 
00145         const int4 bin_id = make_int4(
00146             (node_size.x < 1.0e-8f ? 0 : quantize( (center[0] - node_bbox.bmin.x) / node_size.x, BINS )),
00147             (node_size.y < 1.0e-8f ? 0 : quantize( (center[1] - node_bbox.bmin.y) / node_size.y, BINS )),
00148             (node_size.z < 1.0e-8f ? 0 : quantize( (center[2] - node_bbox.bmin.z) / node_size.z, BINS )),
00149             0 );
00150 
00151         objects.bin_ids[idx] = bin_id;
00152 
00153         const uint32 binX = split_id + (BINS * 0 + bin_id.x)*queue.size;
00154         const uint32 binY = split_id + (BINS * 1 + bin_id.y)*queue.size;
00155         const uint32 binZ = split_id + (BINS * 2 + bin_id.z)*queue.size;
00156 
00157         update_bin( bins.bmin + binX, bins.bmax + binX, (int32*)bins.size + binX, bbox );
00158         update_bin( bins.bmin + binY, bins.bmax + binY, (int32*)bins.size + binY, bbox );
00159         update_bin( bins.bmin + binZ, bins.bmax + binZ, (int32*)bins.size + binZ, bbox );
00160     }
