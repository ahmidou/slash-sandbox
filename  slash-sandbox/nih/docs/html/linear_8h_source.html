00034 template <typename T, int N>
00035 Vector<T,N> scale_factor(const Matrix<T,N,N>& A)
00036 {
00037         int i,j;
00038     Vector<T,N> S;
00039         
00040         for (i = 0; i < N; ++i)
00041         {
00042                 S[i] = fabs(A[i][0]);
00043                 for (j = 1; j < N; ++j)
00044                 {
00045             if (S[i] < fabs(A[i][j]))
00046                 S[i] = fabs(A[i][j]);
00047                 }
00048         }
00049         return S;
00050                 
00051 }

00053 template <typename T, int N>
00054 bool pivot_partial(Matrix<T,N,N>& A, const Vector<T,N>& S, Vector<T,N>& B)
00055 {
00056         int i,j;
00057         T temp;
00058         for(j = 0; j < N; ++j)
00059         {
00060                 for(i = j + 1; i < N; ++i)
00061                 {
00062                         if (S[i] == 0)
00063                         {
00064                                 if(B[i] == 0)
00065                     return false; // System doesn´t have a unique solution
00066                                 else 
00067                     return false; // System is incosistent
00068                         }
00069                         if (fabs(A[i][j]/S[i]) > fabs(A[j][j]/S[j]))
00070                         {
00071                 std::swap( A[i], A[j] );
00072                                 temp = B[i];
00073                                 B[i] = B[j];
00074                                 B[j] = temp;
00075                         }
00076                 }
00077                 
00078                 if (A[j][j] == 0)
00079             return false; // System is singular
00080         }
00081         return true;
00082 }

00084 template <typename T, int N>
00085 bool forward_elimination(Matrix<T,N,N>& A, Vector<T,N>& B)
00086 {
00087         int i,j,k;
00088         T m;
00089         
00090         for (k = 0; k < N-1; ++k)
00091         {
00092                 for (i = k + 1; i < N; ++i)
00093                 {
00094                         m = A[i][k] / A[k][k];
00095                         for (j = k + 1; j < N; ++j)
00096                         {
00097                                 A[i][j] -= m * A[k][j];
00098                                 if (i == j && A[i][j] == 0)
00099                     return false; // Singular system
00100                         }
00101                         B[i] -= m * B[k];
00102                 }
00103         }
00104         return true;            
00105 }

00107 template <typename T, int N>
00108 Vector<T,N> back_substitution(Matrix<T,N,N>& A, Vector<T,N>& B)
00109 {
00110         int i,j;
00111         T sum;
00112         Vector<T,N> X;
00113         X[N-1] = B[N-1]/A[N-1][N-1];
00114         for (i = N - 2; i >= 0; --i)
00115         {
00116                 sum = 0;
00117                 for (j = i + 1; j < N; ++j)
00118                 sum += A[i][j] * X[j];
00119                 X[i] = (B[i] - sum) / A[i][i];
00120         }
00121         return X;
00122 }
