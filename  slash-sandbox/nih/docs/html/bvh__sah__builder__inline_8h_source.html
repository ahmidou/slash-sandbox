00257 class Bvh_sah_builder::Predicate

00259 public:
00260     Predicate(int dim) : m_dim( dim ) {}

00262     bool operator() (const Entity& lhs, const Entity& rhs) const
00263     {
00264         const float leftval  = lhs.m_bbox[0][m_dim] + lhs.m_bbox[1][m_dim];
00265         const float rightval = rhs.m_bbox[0][m_dim] + rhs.m_bbox[1][m_dim];
00266 
00267         if (leftval == rightval)
00268             return lhs.m_index < rhs.m_index;
00269 
00270         return leftval < rightval;

00278 {

00280     return edge[0] * edge[1] + edge[2] * (edge[0] + edge[1]);

00283 inline bool Bvh_sah_builder::find_largest_axis_split(
00284     const Node& node,
00285     uint32&     pivot,
00286     Bbox_type&  left_bb,
00287     Bbox_type&  right_bb)
00288 {
00289     const uint32 begin = node.m_begin;
00290     const uint32 end   = node.m_end;
00291     const uint32 n_entities = end - begin;
00292     if (m_bboxes.size() < n_entities+1)
00293         m_bboxes.resize( n_entities+1 );
00294     if (m_costs.size() < n_entities+1)
00295         m_costs.resize( n_entities+1 );
00296 
00297     float min_cost = std::numeric_limits<float>::max();
00298     int   min_dim  = -1;
00299 
00300     const Vector_type edge = node.m_bbox[1] - node.m_bbox[0];
00301     const int dim = max_element( edge );
00302 
00303     Predicate predicate( dim );
00304 
00305     m_bboxes[0] = Bbox_type();
00306     m_costs[0]  = 0.0f;
00307 
00308     std::vector<int>& indices = m_indices[dim];
00309 
00310     // Accumulate right hand side bboxes.
00311     for(uint32 i = 1; i <= n_entities; ++i)
00312         m_bboxes[i] = Bbox_type( m_bboxes[i-1], m_entities[ indices[ end - i ] ].m_bbox );
00313 
00314     for(uint32 i = 1; i <= n_entities; ++i)
00315         m_costs[i] = m_costs[i-1] + m_entities[ indices[ end - i ] ].m_cost;
00316 
00317     // Loop over possible splits and find the cheapest one.
00318     Bbox_type lbb, rbb;
00319     float     lc = 0.0f, rc;
00320 
00321     for(uint32 num_left = 1; num_left < n_entities; ++num_left)
00322     {
00323         const int num_right = n_entities - num_left;
00324 
00325         rbb = m_bboxes[num_right];
00326         lbb.insert( m_entities[ indices[ begin + num_left - 1 ] ].m_bbox );
00327 
00328         rc  = m_costs[num_right];
00329         lc += m_entities[ indices[ begin + num_left - 1 ] ].m_cost;
00330 
00331         if (m_force_alignment && (num_left % m_max_leaf_size != 0))
00332             continue;
00333 
00334         const float sa_left  = area( lbb );
00335         const float sa_right = area( rbb );
00336 
00337         const float cost = sa_left*lc + sa_right*rc;
00338 
00339         if(cost < min_cost)
00340         {
00341             min_cost = cost;
00342             min_dim  = dim;
00343             pivot    = begin + num_left;
00344             left_bb  = lbb;
00345             right_bb = rbb;
00346         }
00347     }
00348 
00349     if (min_dim == -1)
00350         return false;
00351 
00352     // Compute the cost for the current node.
00353     const float curr_cost = area( node.m_bbox ) * n_entities;
00354 
00355     // Don't split if the split is more expensive than the current node.
00356     if (m_force_splitting == false && curr_cost * 1.5f < min_cost /*TODO: remove magic constant*/ )
00357         return false;
00358 
00359     // Tag the entities whether they're left or right according to the winner split.
00360     const int* indices_ptr = &m_indices[min_dim][0];
00361     for (uint32 i = begin; i < pivot; ++i)
00362         m_tag_bits[indices_ptr[i]] = false;
00363     for (uint32 i = pivot; i < end; ++i)
00364         m_tag_bits[indices_ptr[i]] = true;
00365 
00366     // Rearrange the indices of the non-winner dimensions to sorted left/right.
00367     for (int d = 1; d <= 2; ++d)
00368     {
00369         const int dim = (min_dim+d) % 3;
00370         int leftpos  = begin;
00371         int rightpos = pivot;
00372 
00373         const int* indices_ptr = &m_indices[dim][0];
00374         for (uint32 i = begin; i < end; ++i)
00375         {
00376             const int index = indices_ptr[i];
00377 
00378             if (m_tag_bits[index])
00379                 m_tmpInt[rightpos++] = index;
00380             else
00381                 m_tmpInt[leftpos++] = index;
00382         }
00383 
00384         // Copy the working array back to indices.
00385         memcpy( &m_indices[dim][begin], &m_tmpInt[begin], sizeof(int)*n_entities );
00386     }
00387     return true;
00388 }
