00073         {
00074             if (p1)
00075             {
00076                 // fetch the Morton code corresponding to the split plane
00077                       Integer code = m_codes[ split_index ];
00078                 const uint32  split_dim = level % 3;
00079 
00080                 // extract the selected coordinate
00081                 Integer split_coord = 0;
00082 
00083                 code >>= level-1;
00084                 code <<= level-1;
00085 
00086                 for (int i = 0; code; i++)
00087                 {
00088                         split_coord |= (((code >> split_dim) & 1u) << i);
00089                     code >>= 3u;
00090                 }
00091 
00092                 // convert to floating point
00093                 const float split_plane = convert( m_bbox[0][split_dim], m_bbox[1][split_dim], split_coord );

00095                 // and output the split node
00096                 m_context.write_node(
00097                     node,
00098                     offset,
00099                     skip_node,
00100                     begin,
00101                     end,

00107             {
00108                 // output a leaf node

00168     typedef cuda::Bintree_gen_context::Split_task Split_task;

00170     const uint32 n_points = uint32( points_end - points_begin );
00171 
00172     m_bbox = bbox;
00173     need_space( m_codes, n_points );
00174     need_space( index, n_points );
00175 
00176     // compute the Morton code for each point
00177     thrust::transform(
00178         points_begin,
00179         points_begin + n_points,
00180         m_codes.begin(),
00181         morton_functor<Integer>( bbox ) );

00183     // setup the point indices, from 0 to n_points-1
00184     thrust::copy(
00185         thrust::counting_iterator<uint32>(0),
00186         thrust::counting_iterator<uint32>(0) + n_points,
00187         index.begin() );
00188 
00189     // sort the indices by Morton code
00190     // TODO: use Duane's library directly here... this is doing shameful allocations!
00191     thrust::sort_by_key(
00192         m_codes.begin(),
00193         m_codes.begin() + n_points,
00194         index.begin() );
00195 
00196     // generate a kd-tree
00197     kd::Kd_context<Integer,OutputTree> bintree_context( tree, thrust::raw_pointer_cast( &m_codes.front() ), m_bbox );
00198 
00199     const uint32 bits = kd::Morton_bits<Integer>::value;
00200 
00201     generate(
00202         m_kd_context,
00203         n_points,
00204         thrust::raw_pointer_cast( &m_codes.front() ),
00205         bits,

00207         false,
00208         bintree_context );

00210     m_leaf_count = m_kd_context.m_leaves;
