00052 class MTRand_int32 { // Mersenne Twister random number generator
00053 public:
00054 // default constructor: uses default seed only if this is the first instance
00055   MTRand_int32() { if (!init) seed(5489UL); init = true; }
00056 // constructor with 32 bit int as seed
00057   MTRand_int32(unsigned long s) { seed(s); init = true; }
00058 // constructor with array of size 32 bit ints as seed
00059   MTRand_int32(const unsigned long* array, int size) { seed(array, size); init = true; }
00060 // the two seed functions
00061   void seed(unsigned long); // seed with 32 bit integer
00062   void seed(const unsigned long*, int size); // seed with array
00063 // overload operator() to make this a generator (functor)
00064   unsigned long operator()() { return rand_int32(); }
00065 // 2007-02-11: made the destructor virtual; thanks "double more" for pointing this out
00066   virtual ~MTRand_int32() {} // destructor
00067 protected: // used by derived classes, otherwise not accessible; use the ()-operator
00068   unsigned long rand_int32(); // generate 32 bit random integer
00069 private:
00070   static const int n = 624, m = 397; // compile time constants
00071 // the variables below are static (no duplicates can exist)
00072   static unsigned long state[n]; // state vector array
00073   static int p; // position in state array
00074   static bool init; // true if init function is called
00075 // private functions used to generate the pseudo random numbers
00076   unsigned long twiddle(unsigned long, unsigned long); // used by gen_state()
00077   void gen_state(); // generate new state
00078 // make copy constructor and assignment operator unavailable, they don't make sense
00079   MTRand_int32(const MTRand_int32&); // copy constructor not defined
00080   void operator=(const MTRand_int32&); // assignment operator not defined
00081 };

00084 inline unsigned long MTRand_int32::twiddle(unsigned long u, unsigned long v) {
00085   return (((u & 0x80000000UL) | (v & 0x7FFFFFFFUL)) >> 1)
00086     ^ ((v & 1UL) ? 0x9908B0DFUL : 0x0UL);
00087 }

00089 inline unsigned long MTRand_int32::rand_int32() { // generate 32 bit random int
00090   if (p == n) gen_state(); // new state vector needed
00091 // gen_state() is split off to be non-inline, because it is only called once
00092 // in every 624 calls and otherwise irand() would become too big to get inlined
00093   unsigned long x = state[p++];
00094   x ^= (x >> 11);
00095   x ^= (x << 7) & 0x9D2C5680UL;
00096   x ^= (x << 15) & 0xEFC60000UL;
00097   return x ^ (x >> 18);
00098 }

00100 // generates double floating point numbers in the half-open interval [0, 1)
00101 class MTRand : public MTRand_int32 {
00102 public:
00103   MTRand() : MTRand_int32() {}
00104   MTRand(unsigned long seed) : MTRand_int32(seed) {}
00105   MTRand(const unsigned long* seed, int size) : MTRand_int32(seed, size) {}
00106   ~MTRand() {}
00107   double operator()() {
00108     return static_cast<double>(rand_int32()) * (1. / 4294967296.); } // divided by 2^32
