00223     {
00224         const float cost1 = compute_sah_cost( bvh, node.get_index() );

00227         const Bbox3f bbox1 = bvh.m_bboxes[ node.get_index()    ];

00230         const Bbox3f bbox = bvh.m_bboxes[ node_index ];
