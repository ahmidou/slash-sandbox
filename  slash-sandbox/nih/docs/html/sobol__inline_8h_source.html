00096     for (unsigned int k = 0; k < s; ++k)
00097         matrix[k] = m[k];
00098 
00099     // the remaining direction numbers are obtained by recurrence
00100     for (unsigned int k = s; k < matrixSize; k++) {
00101         matrix[k] = (matrix[k - s] << s) ^ matrix[k - s];
00102         // iterate over bits of polynomial
00103         for (int i = int(s) - 1, p = a >> 1; i > 0; i--, p >>= 1)
00104         {
00105             if (p & 1)
00106                 matrix[k] ^= (matrix[k - i] << i);
00107         }
00108     }
00109     // k-th column is the binary expansion of m_k / 2^k
00110     for (unsigned int k = 0; k < matrixSize; k++)
00111         matrix[k] <<= matrixSize - k - 1;
00112 
00113     return s;
