00039 // TGA File Header
00041 #pragma pack(push,1) // no alignment
00042 struct TGAHeader
00043 {
00044     unsigned char  identsize;           // size of ID field that follows 18 byte header (0 usually)
00045     unsigned char  colourmaptype;       // type of colour map 0=none, 1=has palette
00046     unsigned char  imagetype;           // type of image 0=none,1=indexed,2=rgb,3=grey,+8=rle packed
00047 
00048     unsigned short colourmapstart;      // first colour map entry in palette
00049     unsigned short colourmaplength;     // number of colours in palette
00050     unsigned char  colourmapbits;       // number of bits per palette entry 15,16,24,32
00051 
00052     unsigned short xstart;              // image x origin
00053     unsigned short ystart;              // image y origin
00054     unsigned short width;               // image width in pixels
00055     unsigned short height;              // image height in pixels
00056     unsigned char  bits;                // image bits per pixel 8,16,24,32
00057     unsigned char  descriptor;          // image descriptor bits (vh flip bits)
00058 };
00059 #pragma pack(pop)
