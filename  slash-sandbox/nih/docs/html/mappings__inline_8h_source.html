00032 // maps a point in spherical coordinates to the unit sphere
00033 inline NIH_HOST NIH_DEVICE Vector3f from_spherical_coords(const Vector2f& uv)
00034 {
00035         return Vector3f(
00036                 cosf(uv[0])*sinf(uv[1]),
00037                 sinf(uv[0])*sinf(uv[1]),
00038                 cosf(uv[1]) );

00040 // computes the spherical coordinates of a 3d point
00041 inline NIH_HOST NIH_DEVICE Vector2f to_spherical_coords(const Vector3f& vec)
00042 {
00043     // to spherical in [0,2pi] (phi) and [0,pi] (theta).
00044     float theta, phi;
00045         if (fabsf(vec[2]) >= 1.0f - 1.0e-5f)
00046                 phi = 0.0f;
00047         else {
00048                 phi = atan2f( vec[1], vec[0] );
00049                 phi = phi<0.0f ? phi+2.0f*float(M_PI) : phi;
00050         }
00051     theta = acosf( vec[2] );
00052         return Vector2f(phi,theta);
00053 }

00055 // seedx, seedy is point on [0,1]^2.  x, y is point on radius 1 disk
00056 inline NIH_HOST NIH_DEVICE Vector2f square_to_unit_disk(const Vector2f seed)
00057 {
00058         float phi, r;
00059 
00060         float a = 2*seed[0] - 1;                // (a,b) is now on [-1,1]^2
00061         float b = 2*seed[1] - 1;
00062 
00063         if (a > -b) {                                   // region 1 or 2
00064            if (a > b) {                                 // region 1, also |a| > |b|
00065                    r = a;
00066                    phi = (M_PIf/4 ) * (b/a);
00067            }
00068            else       {                                 // region 2, also |b| > |a|
00069                    r = b;
00070                    phi = (M_PIf/4) * (2 - (a/b));
00071            }
00072         }
00073         else {                                                  // region 3 or 4
00074            if (a < b) {                                 // region 3, also |a| >= |b|, a != 0
00075                         r = -a;
00076                         phi = (M_PIf/4) * (4 + (b/a));
00077            }
00078            else       {                                 // region 4, |b| >= |a|, but a==0 and b==0 could occur.
00079                         r = -b;
00080                         phi = b != 0 ? (M_PIf/4) * (6 - (a/b)) : 0;
00081            }
00082         }
00083 
00084         return Vector2f(
00085                 r * cosf(phi),
00086                 r * sinf(phi) );
00087 }

00089 // diskx, disky is point on radius 1 disk.  x, y is point on [0,1]^2
00090 inline NIH_HOST NIH_DEVICE Vector2f unit_disk_to_square(const Vector2f disk)
00091 {
00092         float r = sqrtf( disk[0]*disk[0] + disk[1]*disk[1] );
00093         float phi = atan2f( disk[1], disk[0] );
00094         float a, b;
00095         if (phi < -M_PIf/4) phi += 2*M_PIf;     // in range [-pi/4,7pi/4]
00096         if ( phi < M_PIf/4) {                           // region 1
00097                 a = r;
00098                 b = phi * a / (M_PIf/4);
00099         }
00100         else if ( phi < 3*M_PIf/4 ) {           // region 2
00101                 b = r;
00102                 a = -(phi - M_PIf/2) * b / (M_PIf/4);
00103         }
00104         else if ( phi < 5*M_PIf/4 ) {           // region 3
00105                 a = -r;
00106                 b = (phi - M_PIf) * a / (M_PIf/4);
00107         }
00108         else {                                                          // region 4
00109                 b = -r;
00110                 a = -(phi - 3*M_PIf/2) * b / (M_PIf/4);
00111         }
00112 
00113         return Vector2f(
00114                 (a + 1) / 2,
00115                 (b + 1) / 2 );
00116 }

00141 // maps the unit square to the sphere with a uniform distribution
00142 inline NIH_HOST NIH_DEVICE Vector3f uniform_square_to_sphere(const Vector2f& uv)
00143 {
00144         const float cosTheta = uv[1]*2.0f - 1.0f;
00145         const float sinTheta = fast_sqrt(max(1.0f - cosTheta*cosTheta,0.0f));
00146         const float phi = uv[0] * 2.0f * float(M_PI);
00147 
00148         return Vector3f(
00149                 fast_cos(phi)*sinTheta,
00150                 fast_sin(phi)*sinTheta,
00151                 cosTheta );
